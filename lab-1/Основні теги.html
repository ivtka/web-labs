<html> — корінь HTML-документа.
    <head> — для метаданих документа.
    <title> — назва сторінки.
    <style> ‒ тег, що дозволяє писати css в html
    <body> — контент документа
      text="#336699"> — встановити колір тексту
      bgcolor="#FFFFCCC" — встановити колір background
    <a> — посилання
      href="index.html" — гіперпосилання на index.html
    <p> — параграф
      align="center" — встановити по центру
      style="letter-spacing:10pt;" — встановлює відстань між буквами 10pt
    <font size="+2"> — встановити розмір шрифту на +2
    <br> — розрив рядка
    <i> — встановити текст курсивом
    <marquee> — створює текст, що біжить
      width="100%" — встановити ширину по розміру екрану
      height="20" — встановити висоту 20
    <img> — світлина
      alt — показує текст після, якщо 
      title — встановлює текст при наведенні на світлину
      height — висота світлини
      width — ширина світлини
      border — встановлює обрамлення
      align="left" — встановлює обтікання текстом справа
      hspace="20" — відступ 20 по горизонталі
      vspace="20" — відступ 20 по вертикалі
    
      style="text-indent: 100px;" — встановлює відступ для першого рядка 100px
      style="line-height: 200%;" — встановлює область висоти рядка
    <b> — встановити текст жирним
